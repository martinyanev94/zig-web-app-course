const std = @import("std");

const Person = struct {
    name: []const u8,
    age: u32,

    fn greet(self: *Person) void {
        std.debug.print("Hello, my name is {} and I am {} years old.\n", .{self.name, self.age});
    }
};

pub fn main() void {
    var person = Person{
        .name = "Alice",
        .age = 30,
    };
    person.greet();
}
