const std = @import("std");

pub fn divide(x: i32, y: i32) !f64 {
    if (y == 0) {
        return error.DivideByZero; // Return an explicit error
    }
    return @intToFloat(f64, x) / @intToFloat(f64, y);
}

pub fn main() void {
    const result = divide(10, 0);
    switch (result) {
        error.DivideByZero => std.debug.print("Error: Division by zero is not allowed!\n", .{}),
        |value| => std.debug.print("Result: {}\n", .{value}),
    }
}
